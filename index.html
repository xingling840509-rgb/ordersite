        function calculateTotal() {

            const checkboxes = document.querySelectorAll('input[name="mealCheck"]:checked');

            let totalItems = 0;

            let totalPrice = 0;


            checkboxes.forEach((cb) => {

                const mealId = cb.value;

                const price = parseInt(cb.parentNode.querySelector('[data-price]').getAttribute('data-price'));

                const qty = parseInt(document.getElementById(`qty_${mealId}`).value);

                

                totalItems += qty;

                totalPrice += price * qty;

            });


            const summaryText = document.getElementById('summaryText');

            if (totalItems > 0) {

                summaryText.innerHTML = `🛒 您已選擇 <span class="text-amber-600 font-bold">${totalItems}</span> 項餐點，總金額為：<span class="text-xl font-extrabold text-amber-600">$${totalPrice}</span> 元`;

            } else {

                summaryText.innerText = "您尚未選擇任何餐點";

            }

        }
